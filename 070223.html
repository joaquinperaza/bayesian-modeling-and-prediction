\(\tilde{\beta} = \beta + n -\hat{\sum} y:\)

n <- length(y)

alpha <- 1

samples <- matrix(,K,1)
samples[1,] <- 0.01

for (k in 2:K){
  p.old <- samples[k-1,]
  p.try <- runif(1,0,1)
  mh <- (prod(dbinom(y,1,p.try))*dbeta(p.try,alpha,beta)) /
        (prod(dbinom(y,1,p.old))*dbeta(p.old,alpha,beta))
  keep <- ifelse(mh>1,1,rbinom(1,1,mh))
  samples[k,] <- ifelse(keep==1, p.try, p.old)
}
